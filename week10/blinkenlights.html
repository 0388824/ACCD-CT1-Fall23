Arduino Morse Code for 'Hello'

#define ledPin 21

void setup() {
  pinMode(ledPin, OUTPUT); 
}

void loop() {


  // H: ....
  dot(); dot(); dot(); dot();
  shortGap();

  // E: .
  dot();
  shortGap();

  // L: .-..
  dot(); dash(); dot(); dot();
  shortGap();

  // L: .-..
  dot(); dash(); dot(); dot();
  shortGap();

  // O: ---
  dash(); dash(); dash();


  longGap();
}

void dot() {
  digitalWrite(ledPin, HIGH);
  delay(250); 
  digitalWrite(ledPin, LOW);
  delay(250); 
}

void dash() {
  digitalWrite(ledPin, HIGH);
  delay(1000); 
  digitalWrite(ledPin, LOW);
  delay(250); 
}

void shortGap() {
  delay(750); 
}

void longGap() {
  delay(1500); 
}
